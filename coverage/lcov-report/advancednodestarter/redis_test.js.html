const { createClient } = require("redis");
 
const CONNECTION_URL = "redis://127.0.0.1:6379"; // default connection url for redis
 
const client = createClient(CONNECTION_URL); // create new redis server
 
client.hset("en", "name", "mostafa"); // add new key value pairs under the main key "en"
client.hset("ar", "name", "مصطفي"); // add new key value pairs under the main key "ar"
 
// get a value given the main and sub keys
client.hget("en", "name", (err, value) => {
    if (err) {
        throw Error(err);
    }
    console.log(value);
});
